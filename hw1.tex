\problem{
Suppose $\M \models \TA$. Show that the embedding $n \mapsto \Term{n}^\M$ of $\Nat$ into $\M$ is elementary.
}

\problem{
Let $\Lang =\{+\}$, where $+$ is a binary function symbol. Show that there are $2^{\aleph_0}$ many non-isomorphic countable $\Lang$-structures.
}

\problem{
Let $\Lang = \{S,0,1\}$, where $S$ is a unary function symbol (\emph{successor}). Consider the $\Lang$-structure $\mathcal{N} = (\Nat, +1, 0,1)$. Show that up to isomorphism, $\operatorname{Th}(\mathcal{N})$ has only $\aleph_0$ many countable models.
}

\problem{
Verify the following statements:
\begin{enumerate}
	\item If $n = l \cdot k$, then  $\PA^- \vdash \Term{n} = \Term{l}\cdot \Term{k}$.
	\item If $n < k$, then  $\PA^- \vdash \Term{n} <  \Term{k}$.
	\item For all $k \in \Nat$,  $\PA^- \vdash \forall x ( x \leq \Term{k} \to x= \Term{0} \vee x=\Term{1} \vee \dots \vee x=\Term{k})$.
\end{enumerate}
}
